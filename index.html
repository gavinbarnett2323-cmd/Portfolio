institutional_edge_v21.pine 4,334 lines 
//  INSTITUTIONAL EDGE v21 — STRICT BAND-CROSS ONLY
//  6-engine composite scoring + 6-gate entry filter
//
//  THE 6 GATES — all must pass for an entry to fire:
//    1. PRIMARY TREND — Daily uptrend (close>EMA50>EMA200)
//    2. PULLBACK ZONE — Pullback ≥4% from 20-bar high
//    3. REVERSAL CONFIRMATION — Bullish candle + RSI turning up
//    4. VOLUME — relVol > 0.85
//    5. NOT EXTENDED — Not at cycle extreme
//    6. ANTI-CLUSTER — ≥12 bars since last fire

indicator("IE v21 — 6-Engine Composite", overlay=false)

// ─── COMPOSITE SCORING ────────────────────────────
masterComposite = 0.0
masterComposite += w1 * c1_trend       // Trend & Momentum
masterComposite += w2 * c2_smartMoney  // Smart Money & Flow
masterComposite += w3 * c3_volatility  // Vol & Exhaustion
masterComposite += w4 * c4_macro       // Macro & Intermarket
masterComposite += w5 * c5_structure   // Structure & Location
masterComposite += w6 * c6_calendar    // Calendar & Events

// BUY  = composite crosses UP through oversold band
// SELL = composite crosses DOWN through overbought band
buySignal  = ta.crossover(masterComposite, -effOS)
sellSignal = ta.crossunder(masterComposite, effOB)
liquidity_global_dashboard_v4.pine 833 lines 
//  LIQUIDITY GLOBAL DASHBOARD v4 — INSTITUTIONAL-GRADE
//
//  WHY v4 EXISTS — FIXING v3's CIRCULAR REASONING
//  v3 used country ETFs as the "liquidity proxy" — when applied
//  to those ETFs, it just plotted a smoothed Z-score of itself.
//
//  v4 BUILDS LIQUIDITY FROM REAL MACRO DRIVERS:
//    1. Money supply (M2 in USD)
//    2. Policy rate (inverted: low rate = expansionary)
//    3. Currency strength (FX vs USD)
//    4. (US only) Net Liquidity = WALCL - RRP - TGA + Reserves

//  PER-COUNTRY COMPOSITE FORMULAS
//   US       — 0.30·NetLiq + 0.20·M2 + 0.20·Reserves + 0.30·(-RealRate)
//   Eurozone — 0.50·M2_USD + 0.30·(-Rate) + 0.20·FX
//   Japan    — 0.50·M2_USD + 0.30·(-Rate) + 0.20·FX
//   China    — 0.50·M2_USD + 0.30·(-Rate) + 0.20·FX
//   UK       — 0.45·M2_USD + 0.30·(-Rate) + 0.25·FX

indicator("Liquidity Global Dashboard v4", overlay=false)

// ─── GLOBAL TIGHTNESS INDEX ───────────────────────
globalTightness = 0.40*dxyZ + 0.30*realYieldZ + 0.30*vixZ
creditSpread    = ta.ema(hygZ - lqdZ, 10)

// 12-week forward projection via momentum extrapolation
fwdProjection = globalLiq + 12 * ta.change(globalLiq, 4) / 4
ie_narrative_engine_v3.pine 1,012 lines 
//  IE NARRATIVE ENGINE v3.1 — PREDICTIVE BUY CONVICTION
//
//  WHY v2 FAILED:
//  1. Hardcoded "D" in request.security — broke on weekly charts
//  2. "Predictive Score" rewarded Leading quadrant even when
//     momentum was decelerating — effectively calling tops
//  3. Table showed rotation STATE but no actionable direction
//
//  v3 "Buy Conviction Score" ONLY scores upside entry potential:
//    Improving + accelerating = HIGHEST (emerging leader)
//    Lagging + accelerating + volume = HIGH (early reversal)
//    Leading + STILL accelerating = MODERATE (continuation)
//    Leading + decelerating = ZERO (topping — NOT a buy)
//    Weakening = ZERO (actively failing)

indicator("IE Narrative Engine v3.1", overlay=true)

// ─── ACTION SCORING ──────────────────────────────
getAction(conviction) =>
    conviction >= 80 ? "STRONG BUY"  :
    conviction >= 60 ? "BUY"         :
    conviction >= 40 ? "ACCUMULATE"  :
    conviction >= 20 ? "HOLD"        :
                        "AVOID"
